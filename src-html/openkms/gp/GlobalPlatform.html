071        public enum APDUMode {
072                // bit values as expected by EXTERNAL AUTHENTICATE
073                CLR(0x00), MAC(0x01), ENC(0x02), RMAC(0x10);

076                private APDUMode(int value) {this.value = value;}
077
078                public static int getSetValue(EnumSet<APDUMode> s) {
079                        int v = 0;
080                        for (APDUMode m : s) {
081                                v |= m.value;
082                        }
083                        return v;
084                }

170        public boolean select(AID sdAID) throws GPException, CardException {
171                // Try to select ISD without giving the sdAID
172                CommandAPDU command = null;
173                if (sdAID == null )
174                        command = new CommandAPDU(ISO7816.CLA_ISO7816, ISO7816.INS_SELECT, 0x04, 0x00, 256);
175                else
176                        command = new CommandAPDU(ISO7816.CLA_ISO7816, ISO7816.INS_SELECT, 0x04, 0x00, sdAID.getBytes(), 256);
177
178                ResponseAPDU resp = channel.transmit(command);
179
180                if (resp.getSW() == 0x6283) {
181                        printStrictWarning("SELECT ISD returned 6283 - CARD_LOCKED");
182                }
183                if (resp.getSW() == 0x9000 || resp.getSW() == 0x6283) {
184                        // The security domain AID is in FCI.
185                        byte[] fci = resp.getData();
186
187                        // Skip template information and find tag 0x84
188                        short aid_offset = TLVUtils.findTag(fci, TLVUtils.skipTagAndLength(fci, (short) 0, (byte) 0x6F), (byte) 0x84);
189                        int aid_length = TLVUtils.getTagLength(fci, aid_offset);
190
191                        AID detectedAID = new AID(fci, aid_offset + 2, aid_length);
192                        verbose("Auto-detected ISD AID: " + detectedAID);
193                        if (sdAID != null && !detectedAID.equals(sdAID)) {
194                                printStrictWarning("SD AID in FCI does not match the requested AID!");
195                        }
196                        this.sdAID = sdAID == null ? detectedAID : sdAID;
197                        return true;
198                        // TODO: parse the maximum command size as well and use with defaultLoadSize
199                }
200                return false;
201        }

220        public List<KeySet.Key> getKeyInfoTemplate() throws CardException, GPException {
221                // Key Information Template
222                CommandAPDU command = new CommandAPDU(getGPCLA(), ISO7816.INS_GET_DATA, 0x00, 0xE0, 256);
223                ResponseAPDU resp = always_transmit(command);
224
225                if (resp.getSW() == ISO7816.SW_CLA_NOT_SUPPORTED) {
226                        command = new CommandAPDU(ISO7816.CLA_ISO7816, ISO7816.INS_GET_DATA, 0x00, 0xE0, 256);
227                        resp = always_transmit(command);
228                }
229                if (resp.getSW() == ISO7816.SW_NO_ERROR) {
230                        return GlobalPlatformData.get_key_template_list(resp.getData(), SHORT_0);
231                } else {
232                        verbose("GET DATA(Key Information Template) not supported");
233                }
234                return GlobalPlatformData.get_key_template_list(null, SHORT_0);
235        }

237        public byte[] fetchCardData() throws CardException, GPException {
238                // Card data
239                CommandAPDU command = new CommandAPDU(getGPCLA(), ISO7816.INS_GET_DATA, 0x00, 0x66, 256);
240                ResponseAPDU resp = always_transmit(command);
241                if (resp.getSW() == 0x6A86) {
242                        verbose("GET DATA(CardData) not supported, Open Platform 2.0.1 card? " + GPUtils.swToString(resp.getSW()));
243                        return null;
244                } else if (resp.getSW() == 0x9000) {
245                        return resp.getData();
246                }
247                return null;
248        }

250        public void discoverCardProperties() throws CardException, GPException {
251
252                // Key Information Template
253                List<Key> key_templates = getKeyInfoTemplate();
254                if (key_templates != null && key_templates.size() > 0) {
255                        GlobalPlatformData.pretty_print_key_template(key_templates, System.out);
256                }
257
258                System.out.println("***** GET DATA:");
259
260                // Issuer Identification Number (IIN)
261                CommandAPDU command = new CommandAPDU(CLA_GP, ISO7816.INS_GET_DATA, 0x00, 0x42, 256);
262                ResponseAPDU resp = channel.transmit(command);
263                if (resp.getSW() == 0x9000) {
264                        System.out.println("IIN " + LoggingCardTerminal.encodeHexString(resp.getData()));
265                } else {
266                        System.out.println("GET DATA(IIN) not supported");
267                }
268
269                // Card Image Number (CIN)
270                command = new CommandAPDU(CLA_GP, ISO7816.INS_GET_DATA, 0x00, 0x45, 256);
271                resp = channel.transmit(command);
272                if (resp.getSW() == 0x9000) {
273                        System.out.println("CIN " + LoggingCardTerminal.encodeHexString(resp.getData()));
274                } else {
275                        System.out.println("GET DATA(CIN) not supported");
276                }
277
278                // Sequence Counter of the default Key Version Number
279                command = new CommandAPDU(CLA_GP, ISO7816.INS_GET_DATA, 0x00, 0xC1, 256);
280                resp = channel.transmit(command);
281                if (resp.getSW() == 0x9000) {
282                        byte [] ssc = resp.getData();
283                        TLVUtils.expectTag(ssc, SHORT_0, (byte) 0xC1);
284                        System.out.println("SSC " + LoggingCardTerminal.encodeHexString(TLVUtils.getTLVValueAsBytes(ssc, SHORT_0)));
285                } else {
286                        System.out.println("GET DATA(SSC) not supported");
287                }
288                System.out.println("*****");
289        }

291        public byte[] fetchCPLC() throws CardException, GPException {
292                CommandAPDU command = new CommandAPDU(getGPCLA(), ISO7816.INS_GET_DATA, 0x9F, 0x7F, 256);
293                ResponseAPDU resp = always_transmit(command);
294                // If GP CLA fails, try with ISO
295                if (resp.getSW() == ISO7816.SW_CLA_NOT_SUPPORTED) {
296                        command = new CommandAPDU(ISO7816.CLA_ISO7816, ISO7816.INS_GET_DATA, 0x9F, 0x7F, 256);
297                        resp = always_transmit(command);
298                }
299
300                if (resp.getSW() == ISO7816.SW_NO_ERROR) {
301                        return resp.getData();
302                } else {
303                        verbose("GET DATA(CPLC) returned SW: " + GPUtils.swToString(resp.getSW()));
304                }
305                return null;
306        }

314        /**
315         * Establishes a secure channel to the security domain. The security domain
316         * must have been selected with {@link open open} before.
317         *
318         * @throws IllegalArgumentException
319         *             if the arguments are out of range or the keyset is undefined
320         * @throws CardException
321         *             if some communication problem is encountered.
322         */
323        public void openSecureChannel(KeySet staticKeys, int scpVersion, EnumSet<APDUMode> securityLevel)
324                        throws CardException, GPException {
325
326                if (sdAID == null)
327                        throw new IllegalStateException("No selected ISD!");
328
329                this.staticKeys = staticKeys;
330                KeySet sessionKeys = null;
331
332                // check for diversification
333                if (Arrays.equals(staticKeys.getKey(KeyType.MAC), GlobalPlatformData.defaultKey) && strict) {
334                        if (GlobalPlatformData.suggestDiversification(getCPLC()) != KeyDiversification.NONE
335                                        && !staticKeys.needsDiversity() && staticKeys.getKeyVersion() == 0x00)
336                                printStrictWarning("Card probably requires EMV diversification but no diversification specified!");
337                }
338
339                if ((scpVersion < SCP_ANY) || (scpVersion > SCP_02_1B)) {
340                        throw new IllegalArgumentException("Invalid SCP version.");
341                }
342
343                if ((scpVersion == SCP_02_0A) || (scpVersion == SCP_02_0B) || (scpVersion == SCP_02_1A) || (scpVersion == SCP_02_1B)) {
344                        throw new IllegalArgumentException("Implicit secure channels cannot be initialized explicitly (use the constructor).");
345                }
346
347                // ENC requires MAC
348                if (securityLevel.contains(APDUMode.ENC))
349                        securityLevel.add(APDUMode.MAC);
350
351                // Parameters check done.
352                byte[] rand = new byte[8];
353                SecureRandom sr = new SecureRandom();
354                sr.nextBytes(rand);
355
356                // P1 key version (SCP1)
357                // P2 either key ID (SCP01) or 0 (SCP2)
358                // TODO: use it here for KeyID?
359                CommandAPDU initUpdate = new CommandAPDU(CLA_GP, INS_INITIALIZE_UPDATE, staticKeys.getKeyVersion(), staticKeys.getKeyID(), rand);
360
361                ResponseAPDU response = channel.transmit(initUpdate);
362                int sw = response.getSW();
363
364                // Detect and report locked cards in a more sensible way.
365                if ((sw == ISO7816.SW_SECURITY_STATUS_NOT_SATISFIED) || (sw == ISO7816.SW_AUTHENTICATION_METHOD_BLOCKED)) {
366                        throw new GPException(sw, "INITIALIZE UPDATE failed, card LOCKED?");
367                }
368
369                // Detect all other errors
370                check(response, "INITIALIZE UPDATE failed");
371
372                byte[] update_response = response.getData();
373                if (update_response.length != 28) {
374                        throw new GPException("Wrong INITIALIZE UPDATE response length: " + update_response.length);
375                }
376                // Set default SCP version based on major version
377                if (scpVersion == SCP_ANY) {
378                        scpVersion = update_response[11] == 2 ? SCP_02_15 : SCP_01_05;
379                }
380                scpMajorVersion = (scpVersion < SCP_02_04) ? 1 : 2;
381                if (scpMajorVersion != update_response[11]) {
382                        throw new GPException("Secure Channel Protocol version mismatch: " + scpMajorVersion + " vs " + update_response[11]);
383                }
384                // Remove RMAC if SCP01 TODO: this should be generic sanitizer somewhere
385                if (scpMajorVersion == 1)
386                        securityLevel.remove(APDUMode.RMAC);
387
388                verbose("Using SCP0" + scpMajorVersion + " with static version " + (update_response[10] & 0xff) + " keys: " + staticKeys);
389
390                // Only diversify default key sets that require it.
391                if ((staticKeys.getKeyVersion() == 0) || (staticKeys.getKeyVersion() == 255)) {
392                        if (staticKeys.needsDiversity()) {
393                                staticKeys.diversify(update_response);
394                                verbose("Diversififed keys: " + staticKeys);
395                        }
396                }
397
398                // If using explicit key version, it must match.
399                if ((staticKeys.getKeyVersion() > 0) && ((update_response[10] & 0xff) != staticKeys.getKeyVersion())) {
400                        throw new GPException("Key set mismatch.");
401                }
402
403                if (scpMajorVersion == 1) {
404                        sessionKeys = deriveSessionKeysSCP01(staticKeys, rand, update_response);
405                } else if (scpMajorVersion == 2) {
406                        byte [] seq = Arrays.copyOfRange(update_response, 12, 14);
407                        sessionKeys = deriveSessionKeysSCP02(staticKeys, seq, false);
408                } else {
409                        throw new GPException("SCP03 is still unexplored");
410                }
411                verbose("Session keys: " + sessionKeys);
412
413
414                ByteArrayOutputStream bo = new ByteArrayOutputStream();
415
416                try {
417                        bo.write(rand);
418                        bo.write(update_response, 12, 8);
419                } catch (IOException ioe) {
420                        throw new RuntimeException(ioe);
421                }
422
423                byte[] myCryptogram = GPUtils.mac_3des(sessionKeys.getKey(KeyType.ENC), GPUtils.pad80(bo.toByteArray()), iv_null_bytes);
424
425                byte[] cardCryptogram = new byte[8];
426                System.arraycopy(update_response, 20, cardCryptogram, 0, 8);
427                if (!Arrays.equals(cardCryptogram, myCryptogram)) {
428                        throw new GPException("Card cryptogram invalid.\nExp: " + GPUtils.byteArrayToString(cardCryptogram) + "\nAct: "+GPUtils.byteArrayToString(myCryptogram));
429                }
430
431                try {
432                        bo.reset();
433                        bo.write(update_response, 12, 8);
434                        bo.write(rand);
435                } catch (IOException ioe) {
436                        throw new RuntimeException(ioe);
437                }
438
439                byte[] authData = GPUtils.mac_3des(sessionKeys.getKey(KeyType.ENC), GPUtils.pad80(bo.toByteArray()), iv_null_bytes);
440
441                wrapper = new SecureChannelWrapper(sessionKeys, scpVersion, EnumSet.of(APDUMode.MAC), null, null);
442
443                int P1 = APDUMode.getSetValue(securityLevel);
444                CommandAPDU externalAuthenticate = new CommandAPDU(CLA_MAC, ISO7816.INS_EXTERNAL_AUTHENTICATE_82, P1, 0, authData);
445                response = transmit(externalAuthenticate);
446
447                check(response, "External authenticate failed");
448
449                wrapper.setSecurityLevel(securityLevel);
450                if (securityLevel.contains(APDUMode.RMAC)) {
451                        wrapper.ricv = new byte[8];
452                        System.arraycopy(wrapper.icv, 0, wrapper.ricv, 0, 8);
453                }
454        }

456        private KeySet deriveSessionKeysSCP01(KeySet staticKeys, byte[] hostRandom, byte[] cardResponse) {
457                byte[] derivationData = new byte[16];
458
459                System.arraycopy(cardResponse, 16, derivationData, 0, 4);
460                System.arraycopy(hostRandom, 0, derivationData, 4, 4);
461                System.arraycopy(cardResponse, 12, derivationData, 8, 4);
462                System.arraycopy(hostRandom, 4, derivationData, 12, 4);
463                KeySet sessionKeys = new KeySet();
464
465                try {
466                        Cipher cipher = Cipher.getInstance("DESede/ECB/NoPadding");
467                        for (KeyType v: KeyType.values()) {
468                                if (v == KeyType.RMAC) continue;
469                                cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(staticKeys.get3DES(v), "DESede"));
470                                sessionKeys.setKey(v, cipher.doFinal(derivationData));
471                        }
472                } catch (BadPaddingException e) {
473                        throw new RuntimeException("Session keys calculation failed.", e);
474                } catch (NoSuchAlgorithmException e) {
475                        throw new RuntimeException("Session keys calculation failed.", e);
476                } catch (NoSuchPaddingException e) {
477                        throw new RuntimeException("Session keys calculation failed.", e);
478                } catch (InvalidKeyException e) {
479                        throw new RuntimeException("Session keys calculation failed.", e);
480                } catch (IllegalBlockSizeException e) {
481                        throw new RuntimeException("Session keys calculation failed.", e);
482                }
483
484                // KEK is the same
485                sessionKeys.setKey(KeyType.KEK, staticKeys.getKey(KeyType.KEK));
486                return sessionKeys;
487        }

489        private KeySet deriveSessionKeysSCP02(KeySet staticKeys, byte[] sequence, boolean implicitChannel) throws CardException {
490                KeySet sessionKeys = new KeySet();
491
492                try {
493                        Cipher cipher = Cipher.getInstance("DESede/CBC/NoPadding");
494
495                        byte[] derivationData = new byte[16];
496                        System.arraycopy(sequence, 0, derivationData, 2, 2);
497
498                        byte[] constantMAC = new byte[] { (byte) 0x01, (byte) 0x01 };
499                        System.arraycopy(constantMAC, 0, derivationData, 0, 2);
500                        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(staticKeys.get3DES(KeyType.MAC), "DESede"), iv_null);
501                        sessionKeys.setKey(KeyType.MAC, cipher.doFinal(derivationData));
502
503                        // TODO: is this correct? - increment by one for all other than C-MAC
504                        if (implicitChannel) {
505                                TLVUtils.buffer_increment(derivationData, (short)2, (short)2);
506                        }
507
508
509                        byte[] constantRMAC = new byte[] { (byte) 0x01, (byte) 0x02 };
510                        System.arraycopy(constantRMAC, 0, derivationData, 0, 2);
511                        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(staticKeys.get3DES(KeyType.MAC), "DESede"), iv_null);
512                        sessionKeys.setKey(KeyType.RMAC, cipher.doFinal(derivationData));;
513
514
515                        byte[] constantENC = new byte[] { (byte) 0x01, (byte) 0x82 };
516                        System.arraycopy(constantENC, 0, derivationData, 0, 2);
517                        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(staticKeys.get3DES(KeyType.ENC), "DESede"), iv_null);
518                        sessionKeys.setKey(KeyType.ENC, cipher.doFinal(derivationData));
519
520                        byte[] constantDEK = new byte[] { (byte) 0x01, (byte) 0x81 };
521                        System.arraycopy(constantDEK, 0, derivationData, 0, 2);
522                        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(staticKeys.get3DES(KeyType.KEK), "DESede"), iv_null);
523                        sessionKeys.setKey(KeyType.KEK, cipher.doFinal(derivationData));
524
525                } catch (BadPaddingException e) {
526                        throw new RuntimeException("Session keys calculation failed.", e);
527                } catch (NoSuchAlgorithmException e) {
528                        throw new RuntimeException("Session keys calculation failed.", e);
529                } catch (NoSuchPaddingException e) {
530                        throw new RuntimeException("Session keys calculation failed.", e);
531                } catch (InvalidKeyException e) {
532                        throw new RuntimeException("Session keys calculation failed.", e);
533                } catch (IllegalBlockSizeException e) {
534                        throw new RuntimeException("Session keys calculation failed.", e);
535                } catch (InvalidAlgorithmParameterException e) {
536                        throw new RuntimeException("Session keys calculation failed.", e);
537                }
538                return sessionKeys;
539
540        }

568        private void loadCapFile(CapFile cap, boolean includeDebug, boolean separateComponents, boolean loadParam, boolean useHash)
569                        throws GPException, CardException {
570
571                if (getRegistry().allAIDs().contains(cap.getPackageAID())) {
572                        printStrictWarning("Package with AID " + cap.getPackageAID() + " is already present on card");
573                }
574                byte[] hash = useHash ? cap.getLoadFileDataHash(includeDebug) : new byte[0];
575                int len = cap.getCodeLength(includeDebug);
576                byte[] loadParams = loadParam ? new byte[] { (byte) 0xEF, 0x04, (byte) 0xC6, 0x02, (byte) ((len & 0xFF00) >> 8),
577                                (byte) (len & 0xFF) } : new byte[0];
578
579                ByteArrayOutputStream bo = new ByteArrayOutputStream();
580
581                try {
582                        bo.write(cap.getPackageAID().getLength());
583                        bo.write(cap.getPackageAID().getBytes());
584
585                        bo.write(sdAID.getLength());
586                        bo.write(sdAID.getBytes());
587
588                        bo.write(hash.length);
589                        bo.write(hash);
590
591                        bo.write(loadParams.length);
592                        bo.write(loadParams);
593                        bo.write(0);
594                } catch (IOException ioe) {
595                        throw new RuntimeException(ioe);
596                }

611         * Install an applet and make it selectable. The package and applet AID must
612         * be present (ie. non-null). If one of the other parameters is null
613         * sensible defaults are chosen. If installation parameters are used, they
614         * must be passed in a special format, see parameter description below.
615         * <P>
616         * Before installation the package containing the applet must be loaded onto
617         * the card, see {@link #loadCapFile loadCapFile}.
618         * <P>
619         * This method installs just one applet. Call it several times for packages
620         * containing several applets.
621         *
622         * @param packageAID
623         *            the package that containing the applet
624         * @param appletAID
625         *            the applet to be installed
626         * @param instanceAID
627         *            the applet AID passed to the install method of the applet,
628         *            defaults to {@code packageAID} if null
629         * @param privileges
630         *            privileges encoded as byte
631         * @param installParams
632         *            tagged installation parameters, defaults to {@code 0xC9 00}
633         *            (ie. no installation parameters) if null, if non-null the
634         *            format is {@code 0xC9 len data...}
635         */
636        public void installAndMakeSelecatable(AID packageAID, AID appletAID, AID instanceAID, byte privileges, byte[] installParams,
637                        byte[] installToken) throws GPException, CardException {
638
639                if (instanceAID == null) {
640                        instanceAID = appletAID;
641                }
642                if (getRegistry().allAppletAIDs().contains(instanceAID)) {
643                        printStrictWarning("Applet with instance AID " + instanceAID + " is already present on card");
644                }
645                if (installParams == null) {
646                        installParams = new byte[] { (byte) 0xC9, 0x00 };
647                }
648                if (installToken == null) {
649                        installToken = new byte[0];
650                }
651                ByteArrayOutputStream bo = new ByteArrayOutputStream();
652                try {
653                        bo.write(packageAID.getLength());
654                        bo.write(packageAID.getBytes());
655
656                        bo.write(appletAID.getLength());
657                        bo.write(appletAID.getBytes());
658
659                        bo.write(instanceAID.getLength());
660                        bo.write(instanceAID.getBytes());
661
662                        bo.write(1);
663                        bo.write(privileges);
664
665                        bo.write(installParams.length);
666                        bo.write(installParams);
667
668                        bo.write(installToken.length);
669                        bo.write(installToken);
670                } catch (IOException ioe) {
671                        throw new RuntimeException(ioe);
672                }
673
674                CommandAPDU install = new CommandAPDU(CLA_GP, INS_INSTALL, 0x0C, 0x00, bo.toByteArray());
675                ResponseAPDU response = transmit(install);
676                check(response, "Install for Install and make selectable failed");
677                dirty = true;
678        }

744        private byte[] encodeKey(KeySet.Key key, byte []kek, boolean withCheck) {
745                try {
746                        ByteArrayOutputStream bytearrayoutputstream = new ByteArrayOutputStream();
747                        // Only DES at the moment
748                        bytearrayoutputstream.write(0x80);
749                        // Length
750                        bytearrayoutputstream.write(16);
751                        // Encrypt key with KEK
752                        Cipher cipher;
753                        cipher = Cipher.getInstance("DESede/ECB/NoPadding");
754                        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(GPUtils.getKey(kek, 24), "DESede"));
755                        bytearrayoutputstream.write(cipher.doFinal(key.getValue(), 0, 16));
756                        if (withCheck) {
757                                // key check value, 3 bytes with new key over null bytes
758                                bytearrayoutputstream.write(3);
759                                SecretKeySpec ky = new SecretKeySpec(GPUtils.getKey(key.getValue(), 24), "DESede");
760                                cipher.init(Cipher.ENCRYPT_MODE, ky);
761                                byte check[] = cipher.doFinal(iv_null_bytes);
762                                bytearrayoutputstream.write(check, 0, 3);
763                        } else {
764                                bytearrayoutputstream.write(0);
765                        }
766                        return bytearrayoutputstream.toByteArray();
767                } catch (IOException e) {
768                        throw new RuntimeException(e);
769                } catch (InvalidKeyException e) {
770                        throw new RuntimeException(e);
771                } catch (IllegalBlockSizeException e) {
772                        throw new RuntimeException(e);
773                } catch (BadPaddingException e) {
774                        throw new RuntimeException(e);
775                } catch (NoSuchAlgorithmException e) {
776                        throw new RuntimeException(e);
777                } catch (NoSuchPaddingException e) {
778                        throw new RuntimeException(e);
779                }
780        }

783        public void putKeys(List<KeySet.Key> keys, boolean replace) throws GPException, CardException {
784                if (keys.size() < 1 || keys.size() > 3)
785                        throw new IllegalArgumentException("Can add keys up to 3 at a time");
786                if (keys.size() > 1) {
787                        for (int i = 1; i < keys.size(); i++) {
788                                if (keys.get(i-1).getID() != keys.get(i).getID() -1)
789                                        throw new IllegalArgumentException("Key ID-s of multiple keys must be sequential!");
790                        }
791                }
792                // Check if factory keys
793                List<Key> tmpl = getKeyInfoTemplate();
794                if ((tmpl.get(0).getVersion() < 1 || tmpl.get(0).getVersion() > 0x7F) && replace) {
795                        printStrictWarning("Trying to replace factory keys? Is this a virgin card?");
796                }
797
798                int P1 = 0x00; // New key in single command unless replace
799                if (replace)
800                        P1 = keys.get(0).getVersion();
801
802                int P2 = keys.get(0).getID();
803                if (keys.size() > 1)
804                        P2 |= 0x80;
805
806                ByteArrayOutputStream bo = new ByteArrayOutputStream();
807                try {
808                        // New key version
809                        bo.write(keys.get(0).getVersion());
810                        // Key data
811                        for (Key k: keys) {
812                                if (scpMajorVersion == 1) {
813                                        bo.write(encodeKey(k, staticKeys.get3DES(KeyType.KEK), true));
814                                } else if (scpMajorVersion == 2) {
815                                        bo.write(encodeKey(k, wrapper.sessionKeys.get3DES(KeyType.KEK), true));
816                                } else
817                                        throw new IllegalStateException("Unknown SCP version: " + scpMajorVersion);
818                        }
819                } catch (IOException e) {
820                        throw new RuntimeException(e);
821                }
822
823                CommandAPDU command = new CommandAPDU(CLA_GP, INS_PUT_KEY, P1, P2, bo.toByteArray());
824                ResponseAPDU response = transmit(command);
825                check(response,"PUT KEY failed");
826        }

829        private byte[] getConcatenatedStatus(int p1, byte[] data) throws CardException, GPException {
830                CommandAPDU getStatus = new CommandAPDU(CLA_GP, INS_GET_STATUS, p1, 0x00, data);
831                ResponseAPDU response = transmit(getStatus);
832                int sw = response.getSW();
833                if ((sw != ISO7816.SW_NO_ERROR) && (sw != 0x6310)) {
834                        return response.getData(); // Should be empty
835                }
836                ByteArrayOutputStream bo = new ByteArrayOutputStream();
837                try {
838                        bo.write(response.getData());
839
840                        while (response.getSW() == 0x6310) {
841                                getStatus = new CommandAPDU(CLA_GP, INS_GET_STATUS, p1, 0x01, data);
842                                response = transmit(getStatus);
843
844                                bo.write(response.getData());
845
846                                sw = response.getSW();
847                                if ((sw != ISO7816.SW_NO_ERROR) && (sw != 0x6310)) {
848                                        throw new CardException("Get Status failed, SW: " + GPUtils.swToString(sw));
849                                }
850                        }
851                } catch (IOException e) {
852                        throw new RuntimeException(e);
853                }
854                return bo.toByteArray();
855        }

858        /**
859         * Get card status. Perform all possible variants of the get status command
860         * and return all entries reported by the card in an AIDRegistry.
861         *
862         * @return registry with all entries on the card
863         * @throws CardException
864         *             in case of communication errors
865         * @throws GPException
866         */
867        private AIDRegistry getStatus() throws CardException, GPException {
868                AIDRegistry registry = new AIDRegistry();
869                int[] p1s = { 0x80, 0x40 };
870                for (int p1 : p1s) {
871                        // parse data no sub-AID
872                        int index = 0;
873                        byte[] data = getConcatenatedStatus(p1, new byte[] { 0x4F, 0x00 });
874                        while (index < data.length) {
875                                int len = data[index++];
876                                AID aid = new AID(data, index, len);
877                                index += len;
878                                int life_cycle = data[index++];
879                                int privileges = data[index++];
880
881                                AIDRegistryEntry.Kind kind = AIDRegistryEntry.Kind.IssuerSecurityDomain;
882                                if (p1 == 0x40) {
883                                        if ((privileges & 0x80) == 0) {
884                                                kind = AIDRegistryEntry.Kind.Application;
885                                        } else {
886                                                kind = AIDRegistryEntry.Kind.SecurityDomain;
887                                        }
888                                }
889
890                                AIDRegistryEntry entry = new AIDRegistryEntry(aid, life_cycle, privileges, kind);
891                                registry.add(entry);
892                        }
893                }
894                // Order is important here, so that ExM info would get to the set later
895                p1s = new int[] { 0x20, 0x10 };
896                for (int p1 : p1s) {
897                        int index = 0;
898                        byte[] data = getConcatenatedStatus(p1, new byte[] { 0x4F, 0x00 });
899                        while (index < data.length) {
900                                int len = data[index++];
901                                AID aid = new AID(data, index, len);
902                                index += len;
903                                AIDRegistryEntry entry = new AIDRegistryEntry(aid, data[index++], data[index++],
904                                                p1 == 0x10 ? AIDRegistryEntry.Kind.ExecutableLoadFilesAndModules : AIDRegistryEntry.Kind.ExecutableLoadFiles);
905                                if (p1 == 0x10) {
906                                        int num = data[index++];
907                                        for (int i = 0; i < num; i++) {
908                                                len = data[index++];
909                                                aid = new AID(data, index, len);
910                                                index += len;
911                                                entry.addExecutableAID(aid);
912                                        }
913                                }
914                                registry.add(entry);
915                        }
916                }
917                return registry;
918        }

965                        if (mac)

967                        if (enc)

971                public void setSCPVersion(int scp) {
972                        // Major version of wrapper
973                        this.scp = 2;
974                        if (scp < SCP_02_04) {
975                                this.scp = 1;
976                        }
977
978                        // modes
979                        if ((scp == SCP_01_15) || (scp == SCP_02_14) || (scp == SCP_02_15) || (scp == SCP_02_1A) || (scp == SCP_02_1B)) {
980                                icvEnc = true;
981                        } else {
982                                icvEnc = false;
983                        }
984                        if ((scp == SCP_01_05) || (scp == SCP_01_15) || (scp == SCP_02_04) || (scp == SCP_02_05) || (scp == SCP_02_14) || (scp == SCP_02_15)) {
985                                preAPDU = true;
986                        } else {
987                                preAPDU = false;
988                        }
989                        if ((scp == SCP_02_0A) || (scp == SCP_02_0B) || (scp == SCP_02_1A) || (scp == SCP_02_1B)) {
990                                postAPDU = true;
991                        } else {
992                                postAPDU = false;
993                        }
994                }

996                private byte clearBits(byte b, byte mask) {
997                        return (byte) ((b & ~mask) & 0xFF);
998                }

1000                private byte setBits(byte b, byte mask) {
1001                        return (byte) ((b | mask) & 0xFF);

1004                private CommandAPDU wrap(CommandAPDU command) throws CardException {
1005
1006                        try {
1007                                if (rmac) {
1008                                        rMac.reset();
1009                                        rMac.write(clearBits((byte) command.getCLA(), (byte) 0x07));
1010                                        rMac.write(command.getINS());
1011                                        rMac.write(command.getP1());
1012                                        rMac.write(command.getP2());
1013                                        if (command.getNc() >= 0) {
1014                                                rMac.write(command.getNc());
1015                                                rMac.write(command.getData());
1016                                        }
1017                                }
1018                                if (!mac && !enc) {
1019                                        return command;
1020                                }
1021
1022
1023                                int origCLA = command.getCLA();
1024                                int newCLA = origCLA;
1025                                int origINS = command.getINS();
1026                                int origP1 = command.getP1();
1027                                int origP2 = command.getP2();
1028                                byte[] origData = command.getData();
1029                                int origLc = command.getNc();
1030                                int newLc = origLc;
1031                                byte[] newData = null;
1032                                int le = command.getNe();
1033                                ByteArrayOutputStream t = new ByteArrayOutputStream();
1034
1035                                if (origLc > getBlockSize()) {
1036                                        throw new IllegalArgumentException("APDU too long for wrapping.");
1037                                }
1038
1039                                if (mac) {
1040                                        if (icv == null) {
1041                                                icv = new byte[8];
1042                                        } else if (icvEnc) {
1043                                                Cipher c = null;
1044                                                if (scp == 1) {
1045                                                        c = Cipher.getInstance("DESede/ECB/NoPadding");
1046                                                        c.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(sessionKeys.get3DES(KeyType.MAC), "DESede"));
1047                                                } else {
1048                                                        c = Cipher.getInstance("DES/ECB/NoPadding");
1049                                                        c.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(sessionKeys.getDES(KeyType.MAC), "DES"));
1050                                                }
1051                                                // encrypts the future ICV ?
1052                                                icv = c.doFinal(icv);
1053                                        }
1054
1055                                        if (preAPDU) {
1056                                                newCLA = setBits((byte) newCLA, (byte) 0x04);
1057                                                newLc = newLc + 8;
1058                                        }
1059                                        t.write(newCLA);
1060                                        t.write(origINS);
1061                                        t.write(origP1);
1062                                        t.write(origP2);
1063                                        t.write(newLc);
1064                                        t.write(origData);
1065
1066                                        if (scp == 1) {
1067                                                icv = GPUtils.mac_3des(sessionKeys.getKey(KeyType.MAC), GPUtils.pad80(t.toByteArray()), icv);
1068                                        } else {
1069                                                icv = GPUtils.mac_des_3des(sessionKeys.getKey(KeyType.MAC), GPUtils.pad80(t.toByteArray()), icv);
1070                                        }
1071
1072                                        if (postAPDU) {
1073                                                newCLA = setBits((byte) newCLA, (byte) 0x04);
1074                                                newLc = newLc + 8;
1075                                        }
1076                                        t.reset();
1077                                        newData = origData;
1078                                }
1079
1080                                if (enc && (origLc > 0)) {
1081                                        if (scp == 1) {
1082                                                t.write(origLc);
1083                                                t.write(origData);
1084                                                if ((t.size() % 8) != 0) {
1085                                                        byte[] x = GPUtils.pad80(t.toByteArray());
1086                                                        t.reset();
1087                                                        t.write(x);
1088                                                }
1089                                        } else {
1090                                                t.write(GPUtils.pad80(origData));
1091                                        }
1092                                        newLc += t.size() - origData.length;
1093
1094                                        Cipher c = Cipher.getInstance("DESede/CBC/NoPadding");
1095                                        c.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(sessionKeys.get3DES(KeyType.ENC), "DESede"), iv_null);
1096                                        newData = c.doFinal(t.toByteArray());
1097                                        t.reset();
1098                                }
1099                                t.write(newCLA);
1100                                t.write(origINS);
1101                                t.write(origP1);
1102                                t.write(origP2);
1103                                if (newLc > 0) {
1104                                        t.write(newLc);
1105                                        t.write(newData);
1106                                }
1107                                if (mac) {
1108                                        t.write(icv);
1109                                }
1110                                if (le > 0) {
1111                                        t.write(le);
1112                                }
1113                                CommandAPDU wrapped = new CommandAPDU(t.toByteArray());
1114                                return wrapped;
1115                        } catch (IOException e) {
1116                                throw new RuntimeException("APDU wrapping failed", e);
1117                        } catch (NoSuchAlgorithmException e) {
1118                                throw new RuntimeException("APDU wrapping failed", e);
1119                        } catch (NoSuchPaddingException e) {
1120                                throw new RuntimeException("APDU wrapping failed", e);
1121                        } catch (InvalidKeyException e) {
1122                                throw new RuntimeException("APDU wrapping failed", e);
1123                        } catch (InvalidAlgorithmParameterException e) {
1124                                throw new RuntimeException("APDU wrapping failed", e);
1125                        } catch (IllegalBlockSizeException e) {
1126                                throw new RuntimeException("APDU wrapping failed", e);
1127                        } catch (BadPaddingException e) {
1128                                throw new RuntimeException("APDU wrapping failed", e);
1129                        }
1130                }

1132                private ResponseAPDU unwrap(ResponseAPDU response) throws GPException {
1133                        if (rmac) {
1134                                if (response.getData().length < 8) {
1135                                        throw new RuntimeException("Wrong response length (too short).");
1136                                }
1137                                int respLen = response.getData().length - 8;
1138                                rMac.write(respLen);
1139                                rMac.write(response.getData(), 0, respLen);
1140                                rMac.write(response.getSW1());
1141                                rMac.write(response.getSW2());
1142
1143                                ricv = GPUtils.mac_des_3des(sessionKeys.getKey(KeyType.RMAC), GPUtils.pad80(rMac.toByteArray()), ricv);
1144
1145                                byte[] actualMac = new byte[8];
1146                                System.arraycopy(response.getData(), respLen, actualMac, 0, 8);
1147                                if (!Arrays.equals(ricv, actualMac)) {
1148                                        throw new GPException("RMAC invalid.");
1149                                }
1150                                ByteArrayOutputStream o = new ByteArrayOutputStream();
1151                                o.write(response.getBytes(), 0, respLen);
1152                                o.write(response.getSW1());
1153                                o.write(response.getSW2());
1154                                response = new ResponseAPDU(o.toByteArray());
1155                        }
1156                        return response;
1157                }
