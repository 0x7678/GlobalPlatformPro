031        public static String byteArrayToReadableString(byte[] array) {
032                if (array == null) {
033                        return "(null)";
034                }
035                StringBuffer s = new StringBuffer();
036                for (int i = 0; i < array.length; i++) {
037                        char c = (char) array[i];
038                        s.append(((c >= 0x20) && (c < 0x7f)) ? (c) : ("."));
039                }
040                return "|" + s.toString() + "|";
041        }
042
043        public static String byteArrayToString(byte[] a) {
044                return LoggingCardTerminal.encodeHexString(a);
045        }
046
047        public static byte[] stringToByteArray(String s) {
048                s = s.toLowerCase().replaceAll(" ", "").replaceAll(":", "");
049                s = s.replaceAll("0x", "").replaceAll("\n", "").replaceAll("\t", "");
050                return LoggingCardTerminal.decodeHexString(s);
051        }

057        private static byte[] pad80(byte[] text, int offset, int length, int blocksize) {
058                if (length == -1) {
059                        length = text.length - offset;
060                }
061                int totalLength = length;
062                for (totalLength++; (totalLength % blocksize) != 0; totalLength++) {
063                        ;
064                }
065                int padlength = totalLength - length;
066                byte[] result = new byte[totalLength];
067                System.arraycopy(text, offset, result, 0, length);
068                result[length] = (byte) 0x80;
069                for (int i = 1; i < padlength; i++) {
070                        result[length + i] = (byte) 0x00;
071                }
072                return result;
073        }
074
075        public static byte[] pad80(byte[] text, int blocksize) {
076                return pad80(text, 0, text.length, blocksize);
077        }

079        public static byte[] mac_3des(byte[] key, byte[] text, byte[] cv)  {
080                return mac_3des(key, text, 0, text.length, cv);
081        }

083        private static byte[] mac_3des(byte[] key, byte[] text, int offset, int length, byte[] iv) {
084                if (length == -1) {
085                        length = text.length - offset;
086                }
087
088                try {
089                        Cipher cipher = Cipher.getInstance("DESede/CBC/NoPadding");
090                        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(KeySet.getKey(key, 24), "DESede"), new IvParameterSpec(iv));
091                        byte[] result = new byte[8];
092                        byte[] res = cipher.doFinal(text, offset, length);
093                        System.arraycopy(res, res.length - 8, result, 0, 8);
094                        return result;
095                } catch (Exception e) {
096                        throw new RuntimeException("MAC computation failed.", e);
097                }
098        }

100        public static byte[] mac_des_3des(byte[] key, byte[] text, byte[] iv) {
101                return mac_des_3des(key, text, 0, text.length, iv);
102        }

104        private static byte[] mac_des_3des(byte[] key, byte[] text, int offset, int length, byte[] iv) {
105                if (length == -1) {
106                        length = text.length - offset;
107                }
108
109                try {
110
111                        Cipher cipher1 = Cipher.getInstance("DES/CBC/NoPadding");
112                        cipher1.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(KeySet.getKey(key, 8), "DES"), new IvParameterSpec(iv));
113                        Cipher cipher2 = Cipher.getInstance("DESede/CBC/NoPadding");
114                        cipher2.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(KeySet.getKey(key, 24), "DESede"), new IvParameterSpec(iv));
115
116                        byte[] result = new byte[8];
117                        byte[] temp;
118
119                        if (length > 8) {
120                                temp = cipher1.doFinal(text, offset, length - 8);
121                                System.arraycopy(temp, temp.length - 8, result, 0, 8);
122                                cipher2.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(KeySet.getKey(key, 24), "DESede"), new IvParameterSpec(result));
123                        }
124                        temp = cipher2.doFinal(text, (offset + length) - 8, 8);
125                        System.arraycopy(temp, temp.length - 8, result, 0, 8);
126                        return result;
127                } catch (Exception e) {
128                        e.printStackTrace();
129                        throw new RuntimeException("MAC computation failed.", e);
130                }
131        }
