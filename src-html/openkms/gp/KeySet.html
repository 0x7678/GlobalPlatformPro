114                if (value.length < 16)
115                        throw new IllegalArgumentException("Key must be at least 16 bytes");
116                switch (type) {
117                        case ENC:
118                                enc_key = value;
119                                break;
120                        case MAC:
121                                mac_key = value;
122                                break;
123                        case KEK:
124                                kek_key = value;
125                                break;
126                        case RMAC:
127                                rmac_key = value;
128                                break;
129                        default:
130                                break;
131                }

157        private byte[] get3DES(KeyType type) {
158                byte[] key24 = new byte[24];
159                System.arraycopy(getKey(type), 0, key24, 0, 16);
160                System.arraycopy(getKey(type), 0, key24, 16, 8);
161                return key24;
162        }
163
164        private byte[] getDES(KeyType type) {
165                byte[] key8 = new byte[8];
166                System.arraycopy(getKey(type), 0, key8, 0, 8);
167                return key8;
168        }

177        public boolean needsDiversity() {
178                return diversification != KeyDiversification.NONE && !diversified;
179        }
180        public void diversify(byte[] initialize_update_response) {
181                if (diversified || diversification == KeyDiversification.NONE) {
182                        throw new IllegalStateException("Already diversified or not needed!");
183                }
184
185                try {
186                        Cipher cipher = Cipher.getInstance("DESede/ECB/NoPadding");
187                        byte[] data = new byte[16];
188                        for (KeyType v : KeyType.values()) {
189                                if (v == KeyType.RMAC)
190                                        continue;
191
192                                // shift around and fill initialize update data as required.
193                                if (diversification == KeyDiversification.VISA2) {
194                                        fillVisa(data, initialize_update_response, v);
195                                } else if (diversification == KeyDiversification.EMV) {
196                                        fillEmv(data, initialize_update_response, v);
197                                }
198
199                                // Encrypt with current master key
200                                cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(get3DES(v), "DESede"));
201
202                                // Replace the key
203                                setKey(v, cipher.doFinal(data));
204                        }
205
206                        diversified = true;
207                } catch (BadPaddingException e) {
208                        throw new RuntimeException("Diversification failed.", e);
209                } catch (NoSuchAlgorithmException e) {
210                        throw new RuntimeException("Diversification failed.", e);
211                } catch (NoSuchPaddingException e) {
212                        throw new RuntimeException("Diversification failed.", e);
213                } catch (InvalidKeyException e) {
214                        throw new RuntimeException("Diversification failed.", e);
215                } catch (IllegalBlockSizeException e) {
216                        throw new RuntimeException("Diversification failed.", e);
217                }
218        }
219
220        private void fillVisa(byte[] data, byte[] init_update_response, KeyType key) {
221                System.arraycopy(init_update_response, 0, data, 0, 2);
222                System.arraycopy(init_update_response, 4, data, 2, 4);
223                data[6] = (byte) 0xF0;
224                data[7] = (byte) key.value;
225                System.arraycopy(init_update_response, 0, data, 8, 2);
226                System.arraycopy(init_update_response, 4, data, 10, 4);
227                data[14] = (byte) 0x0F;
228                data[15] = (byte) key.value;
229        }
230
231        private void fillEmv(byte[] data, byte[] init_update_response, KeyType key) {
232                // 6 rightmost bytes of init update response (which is 10 bytes)
233                System.arraycopy(init_update_response, 4, data, 0, 6);
234                data[6] = (byte) 0xF0;
235                data[7] = (byte) key.value;
236                System.arraycopy(init_update_response, 4, data, 8, 6);
237                data[14] = (byte) 0x0F;
238                data[15] = (byte) key.value;
239        }

263        protected static byte[] getKey(byte[] key, int length) {
264                if (length == 24) {
265                        byte[] key24 = new byte[24];
266                        System.arraycopy(key, 0, key24, 0, 16);
267                        System.arraycopy(key, 0, key24, 16, 8);
268                        return key24;
269                } else {
270                        byte[] key8 = new byte[8];
271                        System.arraycopy(key, 0, key8, 0, 8);
272                        return key8;
273                }
