025import java.io.ByteArrayOutputStream;

038public class CapFile {

040        public static final String[] componentNames = { "Header", "Directory", "Import", "Applet", "Class", "Method", "StaticField", "Export",
041                "ConstantPool", "RefLocation", "Descriptor", "Debug" };
042
043        private final HashMap<String, byte[]> capComponents = new HashMap<String, byte[]>();
044
045        private String packageName = null;
046
047        private AID packageAID = null;
048
049        private final List<AID> appletAIDs = new ArrayList<AID>();
050
051        private final List<byte[]> dapBlocks = new ArrayList<byte[]>();
052
053        private final List<byte[]> loadTokens = new ArrayList<byte[]>();
054
055        private final List<byte[]> installTokens = new ArrayList<byte[]>();
056
057        public CapFile(InputStream in) throws IOException {
058                this(in, null);
059        }

061        public CapFile(InputStream in, String packageName) throws IOException {
062                ZipInputStream zip = new ZipInputStream(in);
063                Map<String, byte[]> entries = getEntries(zip);
064                if (packageName != null) {
065                        packageName = packageName.replace('.', '/') + "/javacard/";
066                } else {
067                        Iterator<? extends String> it = entries.keySet().iterator();
068                        String lookFor = "Header.cap";
069                        while (it.hasNext()) {
070                                String s = it.next();
071                                if (s.endsWith(lookFor)) {
072                                        packageName = s.substring(0, s.lastIndexOf(lookFor));
073                                        break;
074                                }
075                        }
076                }
077
078                // Avoid a possible NPE
079                if (packageName == null) {
080                        throw new RuntimeException("Could not figure out the package name of the applet!");
081                }
082
083                this.packageName = packageName.substring(0, packageName.lastIndexOf("/javacard/")).replace('/', '.');
084                for (String name : componentNames) {
085                        String fullName = packageName + name + ".cap";
086                        byte[] contents = entries.get(fullName);
087                        capComponents.put(name, contents);
088                }
089                List<List<byte[]>> tables = new ArrayList<List<byte[]>>();
090                tables.add(dapBlocks);
091                tables.add(loadTokens);
092                tables.add(installTokens);
093                String[] names = { "dap", "lt", "it" };
094                for (int i = 0; i < names.length; i++) {
095                        int index = 0;
096                        while (true) {
097                                String fullName = "meta-inf/" + packageName.replace('/', '-') + names[i] + (index + 1);
098                                byte[] contents = entries.get(fullName);
099                                if (contents == null) {
100                                        break;
101                                }
102                                tables.get(i).add(contents);
103                                index++;
104                        }
105                }
106                zip.close();
107                byte[] header = capComponents.get("Header");
108                int i = 0;
109                // header[0] should be 1;
110                i++;
111                // header[1] should be 0;
112                i++;
113                // header[2] should be remaining length
114                i++;
115                // header[3, 4, 5, 6] should be magic
116                i += 4;
117                // header[7, 8] should be cap file version
118                i += 2;
119                // header[9] should be flags
120                i++;
121                // header[10,11] should be package version
122                i += 2;
123                // header[12] should be the length of AID
124                int len = header[i++];
125                packageAID = new AID(header, i, len);
126                //GPUtils.debug("package AID: " + packageAID);
127
128                byte[] applet = capComponents.get("Applet");
129                if (applet != null) {
130                        i = 0;
131                        // applet[0] should be 3;
132                        i++;
133                        // applet[1] should be 0;
134                        i++;
135                        // applet[2] should be remaining length
136                        i++;
137                        // header[3] should be number of applets
138                        int num = applet[i++];
139                        for (int j = 0; j < num; j++) {
140                                len = applet[i++];
141                                appletAIDs.add(new AID(applet, i, len));
142                                i += len + 2;
143                        }
144                        //GPUtils.debug("applet AIDs: " + appletAIDs);
145                } else {
146                        //GPUtils.debug("No Applet component.");
147                }
148        }
149
150        private Map<String, byte[]> getEntries(ZipInputStream in) throws IOException {
151                Map<String, byte[]> result = new HashMap<String, byte[]>();
152                while (true) {
153                        ZipEntry entry = in.getNextEntry();
154                        if (entry == null) {
155                                break;
156                        }
157                        if (entry.getName().indexOf("MANIFEST.MF") != -1) {
158                                continue;
159                        }
160                        ByteArrayOutputStream bos = new ByteArrayOutputStream();
161                        byte[] buf = new byte[1024];
162                        int c;
163                        while ((c = in.read(buf)) > 0) {
164                                bos.write(buf, 0, c);
165                        }
166                        result.put(entry.getName(), bos.toByteArray());
167                }
168                return result;
169        }

185        public int getCodeLength(boolean includeDebug) {
186                int result = 0;
187                for (String name : componentNames) {
188                        if (!includeDebug && (name.equals("Debug") || name.equals("Descriptor"))) {
189                                continue;
190                        }
191                        byte[] data = capComponents.get(name);
192                        if (data != null) {
193                                result += data.length;
194                        }
195                }
196                return result;
197        }
198
199        private byte[] createHeader(boolean includeDebug) {
200                int len = getCodeLength(includeDebug);
201                ByteArrayOutputStream bo = new ByteArrayOutputStream();
202                bo.write((byte) 0xC4);
203                if (len < 0x80) {
204                        bo.write((byte) len);
205                } else if (len <= 0xFF) {
206                        bo.write((byte) 0x81);
207                        bo.write((byte) len);
208                } else if (len <= 0xFFFF) {
209                        bo.write((byte) 0x82);
210                        bo.write((byte) ((len & 0xFF00) >> 8));
211                        bo.write((byte) (len & 0xFF));
212                } else {
213                        bo.write((byte) 0x83);
214                        bo.write((byte) ((len & 0xFF0000) >> 16));
215                        bo.write((byte) ((len & 0xFF00) >> 8));
216                        bo.write((byte) (len & 0xFF));
217                }
218                return bo.toByteArray();
219        }

221        public List<byte[]> getLoadBlocks(boolean includeDebug, boolean separateComponents, int blockSize) {
222                List<byte[]> blocks = null;
223
224                if (!separateComponents) {
225                        ByteArrayOutputStream bo = new ByteArrayOutputStream();
226                        try {
227                                bo.write(createHeader(includeDebug));
228                                bo.write(getRawCode(includeDebug));
229                        } catch (IOException ioe) {
230
231                        }
232                        blocks = splitArray(bo.toByteArray(), blockSize);
233                } else {
234                        for (String name : componentNames) {
235                                if (!includeDebug && (name.equals("Debug") || name.equals("Descriptor"))) {
236                                        continue;
237                                }
238
239                                byte[] currentComponent = capComponents.get(name);
240                                if (currentComponent == null) {
241                                        continue;
242                                }
243                                if (name.equals("Header")) {
244                                        ByteArrayOutputStream bo = new ByteArrayOutputStream();
245                                        try {
246                                                bo.write(createHeader(includeDebug));
247                                                bo.write(currentComponent);
248                                        } catch (IOException ioe) {
249
250                                        }
251                                        currentComponent = bo.toByteArray();
252                                }
253                                blocks = splitArray(currentComponent, blockSize);
254                        }
255                }
256                return blocks;
257        }
258
259        private byte[] getRawCode(boolean includeDebug) {
260                byte[] result = new byte[getCodeLength(includeDebug)];
261                short offset = 0;
262                for (String name : componentNames) {
263                        if (!includeDebug && (name.equals("Debug") || name.equals("Descriptor"))) {
264                                continue;
265                        }
266                        byte[] currentComponent = capComponents.get(name);
267                        if (currentComponent == null) {
268                                continue;
269                        }
270                        System.arraycopy(currentComponent, 0, result, offset, currentComponent.length);
271                        offset += currentComponent.length;
272                }
273                return result;
274        }
275
276        public byte[] getLoadFileDataHash(boolean includeDebug) {
277                try {
278                        return MessageDigest.getInstance("SHA1").digest(getRawCode(includeDebug));
279                } catch (NoSuchAlgorithmException e) {
280                        throw new RuntimeException("Not possible", e);
281                }
282        }
283
284        private List<byte[]> splitArray(byte[] array, int blockSize) {
285                List<byte[]> result = new ArrayList<byte[]>();
286                int len = array.length;
287                int offset = 0;
288                int left = len - offset;
289                while (left > 0) {
290                        int currentLen = 0;
291                        if (left >= blockSize) {
292                                currentLen = blockSize;
293                        } else {
294                                currentLen = left;
295                        }
296                        byte[] block = new byte[currentLen];
297                        System.arraycopy(array, offset, block, 0, currentLen);
298                        result.add(block);
299                        left -= currentLen;
300                        offset += currentLen;
301                }
302                return result;
303        }
