023package openkms.gpj;
024
025import java.util.ArrayList;
026import java.util.Iterator;
027import java.util.List;
028
029/**
030 * Convenience class managing a vector of {@link AIDRegistryEntry
031 * AIDRegistryEntries} with search functionality.
032 *
033 * Implements {@code Iterable<AIDRegistryEntry} to permit foreach loops such as
034 * {@code for(AIDRegistryEntry e : registry) ...}.
035 */
036public class AIDRegistry implements Iterable<AIDRegistryEntry> {
037
038        List<AIDRegistryEntry> entries = new ArrayList<AIDRegistryEntry>();
039
040        /**
041         * Add one entry to this registry.
042         *
043         * @param entry
044         */
045        public void add(AIDRegistryEntry entry) {
046                entries.add(entry);
047        }
048
049        /**
050         * Returns an iterator that iterates over all entries in this registry.
051         *
052         * @return iterator
053         */
054        public Iterator<AIDRegistryEntry> iterator() {
055                return entries.iterator();
056        }
057
058
059        /**
060         * Returns a list of all packages in this registry.
061         *
062         * @return a list of all packages
063         */
064        public List<AIDRegistryEntry> allPackages() {
065                List<AIDRegistryEntry> res = new ArrayList<AIDRegistryEntry>();
066                for (AIDRegistryEntry e : entries) {
067                        if (e.isPackage()) {
068                                res.add(e);
069                        }
070                }
071                return res;
072        }
073
074        public List<AID> allPackageAIDs() {
075                List<AID> res = new ArrayList<AID>();
076                for (AIDRegistryEntry e : entries) {
077                        if (e.isPackage()) {
078                                res.add(e.getAID());
079                        }
080                }
081                return res;
082        }
083        public List<AID> allAppletAIDs() {
084                List<AID> res = new ArrayList<AID>();
085                for (AIDRegistryEntry e : entries) {
086                        if (e.isApplet()) {
087                                res.add(e.getAID());
088                        }
089                }
090                return res;
091        }
092        public List<AID> allAIDs() {
093                List<AID> res = new ArrayList<AID>();
094                for (AIDRegistryEntry e : entries) {
095                        res.add(e.getAID());
096                }
097                return res;
098        }
099        /**
100         * Returns a list of all applets in this registry.
101         *
102         * @return a list of all applets
103         */
104        public List<AIDRegistryEntry> allApplets() {
105                List<AIDRegistryEntry> res = new ArrayList<AIDRegistryEntry>();
106                for (AIDRegistryEntry e : entries) {
107                        if (e.isApplet()) {
108                                res.add(e);
109                        }
110                }
111                return res;
112        }
113
114        public AID getDefaultSelectedAID() {
115                for (AIDRegistryEntry e : allApplets()) {
116                        if ((e.getPrivileges() & 0x04) != 0) {
117                                return e.getAID();
118                        }
119                }
120                return null;
121        }
122
123        public AID getDefaultSelectedPackageAID() {
124                AID defaultAID = getDefaultSelectedAID();
125                if (defaultAID != null) {
126                        for (AIDRegistryEntry e : allPackages()) {
127                                if (e.getExecutableAIDs().contains(defaultAID))
128                                        return e.getAID();
129                        }
130                        // Did not get a hit. Loop packages and look for prefixes
131                        for (AIDRegistryEntry e : allPackages()) {
132                                if (defaultAID.toString().startsWith(e.getAID().toString()))
133                                        return e.getAID();
134                        }
135                }
136                return null;
137        }
138}
