001package openkms.gpj;
002
003import java.io.File;
004import java.io.FileInputStream;
005import java.util.Arrays;
006import java.util.EnumSet;
007import java.util.List;
008
009import javax.smartcardio.Card;
010import javax.smartcardio.CardException;
011import javax.smartcardio.CardTerminal;
012import javax.smartcardio.CardTerminals;
013import javax.smartcardio.CardTerminals.State;
014import javax.smartcardio.TerminalFactory;
015
016import joptsimple.OptionException;
017import joptsimple.OptionParser;
018import joptsimple.OptionSet;
019
020
021public class GPJTool {
022
023        private final static String CMD_INFO = "info";
024
025        private final static String CMD_LIST = "list";
026        private final static String CMD_LOCK = "lock";
027        private final static String CMD_INSTALL = "install";
028        private final static String CMD_DELETE = "delete";
029        private final static String CMD_CREATE = "create";
030        private final static String CMD_LOAD = "load";
031        private final static String CMD_UNLOCK = "unlock";
032
033
034        private final static String OPT_DELETEDEPS = "deletedeps";
035        private final static String OPT_DEFAULT = "default";
036        private final static String OPT_CAP = "cap";
037        private final static String OPT_APPLET = "applet";
038        private final static String OPT_PACKAGE = "package";
039        private final static String OPT_INSTANCE = "instance";
040        private final static String OPT_DO_ALL_READERS = "all";
041
042
043        private final static String OPT_CONTINUE = "skip-error";
044        private final static String OPT_RELAX = "relax";
045        private final static String OPT_READER = "reader";
046        private final static String OPT_VERSION = "version";
047        private final static String OPT_SDAID = "sdaid";
048        private final static String OPT_DEBUG = "debug";
049        private final static String OPT_VERBOSE = "verbose";
050        private final static String OPT_REINSTALL = "reinstall";
051
052        private final static String OPT_MODE = "mode";
053
054        private final static String OPT_MAC = "mac";
055        private final static String OPT_ENC = "enc";
056        private final static String OPT_KEK = "kek";
057        private final static String OPT_KEY = "key";
058        private final static String OPT_KEY_VERSION = "keyver";
059        private final static String OPT_KEY_ID = "keyid";
060
061        private final static String OPT_EMV = "emv";
062        private final static String OPT_VISA2 = "visa2";
063
064
065
066
067        public static void main(String[] argv) throws Exception {
068                OptionSet args = null;
069                OptionParser parser = new OptionParser();
070
071                // Generic options
072                parser.acceptsAll(Arrays.asList("h", "help"), "Shows this help string").forHelp();
073                parser.acceptsAll(Arrays.asList("d", OPT_DEBUG), "Show PC/SC and APDU trace");
074                parser.acceptsAll(Arrays.asList("v", OPT_VERBOSE), "Be verbose about operations");
075                parser.acceptsAll(Arrays.asList("r", OPT_READER), "Use specific reader").withRequiredArg();
076                parser.acceptsAll(Arrays.asList("l", CMD_LIST), "List the contents of the card");
077                parser.acceptsAll(Arrays.asList("i", CMD_INFO), "Show information");
078                parser.accepts(OPT_VERSION, "Show information about the program");
079
080                // Special options
081                parser.accepts(OPT_RELAX, "Relaxed error checking");
082                parser.accepts(OPT_DO_ALL_READERS, "Work with multiple readers");
083
084                // Applet operation options
085                parser.accepts(OPT_CAP, "Use a CAP file as source").withRequiredArg().ofType(File.class);
086                parser.accepts(CMD_LOAD, "Load a CAP file").withRequiredArg().ofType(File.class);
087
088                parser.accepts(CMD_INSTALL, "Install applet").withOptionalArg().ofType(File.class);
089                parser.accepts(OPT_DEFAULT, "Indicate Default Selected");
090                parser.accepts(OPT_DELETEDEPS, "Also delete dependencies");
091                parser.accepts(OPT_REINSTALL, "Remove card content during installation");
092
093                parser.accepts(CMD_DELETE, "Delete something").requiredIf(OPT_DELETEDEPS).withOptionalArg().withValuesConvertedBy(GPJToolArgumentMatchers.aid());
094
095                // TODO: require CAP file here, for now (usability)
096                parser.accepts(CMD_CREATE, "Create new instance of an applet").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.aid());
097                parser.accepts(OPT_APPLET, "Applet AID").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.aid());
098                parser.accepts(OPT_PACKAGE, "Package AID").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.aid());
099                parser.accepts(OPT_INSTANCE, "Instance AID").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.aid());
100
101                // Key options
102                parser.accepts(OPT_MAC, "Specify MAC key").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.key());
103                parser.accepts(OPT_ENC, "Specify ENC key").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.key());
104                parser.accepts(OPT_KEK, "Specify KEK key").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.key());
105                parser.accepts(OPT_KEY, "Specify master key").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.key());
106                parser.accepts(CMD_LOCK, "Set new key").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.key());
107                parser.accepts(CMD_UNLOCK, "Set default key").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.key());
108                parser.accepts(OPT_KEY_ID, "Specify key ID").withRequiredArg().ofType(Integer.class);
109                parser.accepts(OPT_KEY_VERSION, "Specify key version").withRequiredArg().ofType(Integer.class);
110
111                // Key diversification and AID options
112                parser.accepts(OPT_EMV, "Use EMV diversification");
113                parser.accepts(OPT_VISA2, "Use VISA2 diversification");
114                parser.accepts(OPT_MODE, "APDU mode to use (mac/enc/clr)").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.mode());;
115
116                parser.accepts(OPT_SDAID, "ISD AID").withRequiredArg().withValuesConvertedBy(GPJToolArgumentMatchers.aid());
117
118
119                // Parse arguments
120                try {
121                        args = parser.parse(argv);
122                        // Try to fetch all values
123                        for (String s: parser.recognizedOptions().keySet()) {args.valueOf(s);}
124                } catch (OptionException e) {
125                        if (e.getCause() != null) {
126                                System.err.println(e.getMessage() + ": " + e.getCause().getMessage());
127                        } else {
128                                System.err.println(e.getMessage());
129                        }
130                        System.err.println();
131                        parser.printHelpOn(System.err);
132                        System.exit(1);
133                }
134
135                // Do the work, based on arguments
136                if (args.has("help")) {
137                        parser.printHelpOn(System.out);
138                        System.exit(0);
139                }
140
141                if (args.has(OPT_VERSION)) {
142                        System.out.println("OpenKMS GlobalPlatform version " + GlobalPlatform.sdk_version);
143                }
144
145                TerminalFactory tf = TerminalManager.getTerminalFactory();
146                CardTerminals terminals = tf.terminals();
147
148                // List terminals if needed
149                if (args.has(OPT_DEBUG)) {
150                        System.out.println("# Detected readers");
151                        for (CardTerminal term : terminals.list()) {
152                                System.out.println((term.isCardPresent() ? "[*] " : "[ ] ") + term.getName());
153                        }
154                }
155
156                // Select terminals to work on
157                List<CardTerminal> do_readers;
158                if (args.has(OPT_READER)) {
159                        CardTerminal t = terminals.getTerminal((String) args.valueOf(OPT_READER));
160                        if (t == null) {
161                                System.err.println("Reader \"" + (String) args.valueOf(OPT_READER) + "\" not found.");
162                                System.exit(1);
163                        }
164                        do_readers = Arrays.asList(t);
165                } else {
166                        do_readers = terminals.list(State.CARD_PRESENT);
167                        if (do_readers.size() > 1 && !args.hasArgument(OPT_DO_ALL_READERS)) {
168                                System.err.println("More than one reader with a card found.");
169                                System.err.println("Run with --"+OPT_DO_ALL_READERS+" to work with all found cards");
170                                System.exit(1);
171                        }
172                }
173
174                // Parameters for opening the secure channel
175                KeySet ks = new KeySet(GlobalPlatformData.defaultKey);
176
177                // Load a CAP file, if specified
178                CapFile cap = null;
179                if (args.has(OPT_CAP)) {
180                        File capfile = (File) args.valueOf(OPT_CAP);
181                        cap = new CapFile(new FileInputStream(capfile));
182                }
183
184                // Work all readers
185                for (CardTerminal reader: do_readers) {
186                        // Wrap with logging
187                        if (args.has(OPT_DEBUG)) {
188                                reader = LoggingCardTerminal.getInstance(reader);
189                        }
190
191                        Card card = null;
192                        try {
193                                // Establish connection
194                                try {
195                                        card = reader.connect("*");
196                                } catch (CardException e) {
197                                        if (args.has(OPT_CONTINUE)) {
198                                                e.printStackTrace();
199                                                continue;
200                                        } else {
201                                                throw e;
202                                        }
203                                }
204
205                                // GlobalPlatform specific
206                                GlobalPlatform gp = new GlobalPlatform(card.getBasicChannel());
207                                if (args.has(OPT_VERBOSE))
208                                        gp.beVerboseTo(System.out);
209
210                                // Be strict unless told otherwise
211                                gp.setStrict(!args.has(OPT_RELAX));
212
213                                if (args.has(CMD_INFO)) {
214                                        System.out.println("Reader: " + reader.getName());
215                                        System.out.println("ATR: " + GPUtils.byteArrayToString(card.getATR().getBytes()));
216                                }
217
218                                // Talk to the card manager
219                                gp.select((AID) args.valueOf(OPT_SDAID));
220
221                                // Fetch some data
222                                if (args.has(CMD_INFO)) {
223                                        System.out.println("***** Card info (not authenticated):");
224                                        GlobalPlatformData.print_card_info(gp);
225                                }
226
227                                // Authenticate, only if needed
228                                if (args.has(CMD_LIST) || args.has(CMD_INSTALL) || args.has(CMD_DELETE) || args.has(CMD_CREATE) || args.has(CMD_LOCK) || args.has(CMD_UNLOCK) ) {
229                                        // MAC by default.
230                                        EnumSet<GlobalPlatform.APDUMode> mode = EnumSet.of(GlobalPlatform.APDUMode.MAC);
231                                        if (args.has(OPT_MODE)) {
232                                                mode.clear();
233                                                mode.add((GlobalPlatform.APDUMode) args.valueOf(OPT_MODE));
234                                        }
235                                        // Possibly brick the card now.
236                                        gp.openSecureChannel(ks, GlobalPlatform.SCP_ANY, mode);
237
238                                        // --delete <aid> or --delete --default
239                                        if (args.has(CMD_DELETE)) {
240                                                if (args.has(OPT_DEFAULT)) {
241                                                        gp.uninstallDefaultSelected(args.has(OPT_DELETEDEPS));
242                                                }
243                                                @SuppressWarnings("unchecked")
244                                                List<AID> aids = (List<AID>) args.valuesOf(CMD_DELETE);
245                                                for (AID aid: aids) {
246                                                        try {
247                                                                gp.deleteAID(aid, args.has(OPT_DELETEDEPS));
248                                                        } catch (GPException e) {
249                                                                if (!gp.getRegistry().allAIDs().contains(aid)) {
250                                                                        System.out.println("Could not delete AID (not present on card): " + aid);
251                                                                } else {
252                                                                        System.out.println("Could not delete AID: " + aid);
253                                                                        if (e.sw == 0x6985) {
254                                                                                System.out.println("TIP: Maybe try with --" + OPT_DELETEDEPS);
255                                                                        }
256                                                                        throw e;
257                                                                }
258                                                        }
259                                                }
260                                        }
261
262                                        // --install <applet.cap>
263                                        if (args.has(CMD_INSTALL)) {
264                                                AID def = gp.getRegistry().getDefaultSelectedAID();
265                                                if (def != null && args.has(OPT_DEFAULT)) {
266                                                        if (args.has(OPT_REINSTALL)) {
267                                                                gp.verbose("Removing current default applet/package");
268                                                                // Remove all instances of default selected app
269                                                                def = gp.getRegistry().getDefaultSelectedPackageAID();
270                                                                gp.deleteAID(def, true); // XXX: What about different instances ?
271                                                        }
272                                                }
273
274                                                File capfile = (File) args.valueOf(CMD_INSTALL);
275                                                CapFile instcap = new CapFile(new FileInputStream(capfile));
276
277                                                // Check if already installed, for some reason
278                                                AID aid = instcap.getAppletAIDs().get(0);
279
280                                                if (gp.getRegistry().allAIDs().contains(aid)) {
281                                                        System.err.println("WARNING: Applet " + aid + " already present on card");
282                                                }
283
284                                                gp.verbose("Installing applet from package " + instcap.getPackageName());
285                                                gp.loadCapFile(instcap);
286                                                // instance will be aid, which is first applet from package
287                                                gp.installAndMakeSelecatable(instcap.getPackageAID(), aid, null, args.has(OPT_DEFAULT) ? (byte) 0x04 : 0x00, null, null);
288                                        }
289
290                                        // --create <aid> (--applet <aid> --package <aid> or --cap <cap>)
291                                        if (args.has(CMD_CREATE)) {
292                                                AID packageAID = null;
293                                                AID appletAID = null;
294                                                // Load from cap if present
295                                                if (cap != null) {
296                                                        packageAID = cap.getPackageAID();
297                                                        appletAID = cap.getAppletAIDs().get(0);
298                                                }
299                                                // override if needed
300                                                packageAID = (AID) args.valueOf(OPT_PACKAGE);
301                                                appletAID = (AID) args.valueOf(OPT_APPLET);
302
303                                                // check
304                                                if (packageAID == null || appletAID == null)
305                                                        throw new IllegalArgumentException("Need --" + OPT_PACKAGE + " and --" + OPT_APPLET + " or --" + OPT_CAP);
306
307                                                // shoot
308                                                AID instanceAID = (AID) args.valueOf(CMD_CREATE);
309                                                gp.installAndMakeSelecatable(packageAID, appletAID, instanceAID, (byte) 0x00, null, null);
310                                        }
311
312                                        // --list
313                                        if (args.has(CMD_LIST)) {
314                                                AIDRegistry registry = gp.getStatus();
315                                                registry = gp.getStatus();
316                                                for (AIDRegistryEntry e : registry) {
317                                                        AID aid = e.getAID();
318                                                        System.out.println("AID: " + GPUtils.byteArrayToString(aid.getBytes()) + " (" + GPUtils.byteArrayToReadableString(aid.getBytes()) + ")");
319                                                        System.out.println("     " + e.getKind().toShortString() + " " + e.getLifeCycleString() + ": " + e.getPrivilegesString());
320
321                                                        for (AID a : e.getExecutableAIDs()) {
322                                                                System.out.println("     " + GPUtils.byteArrayToString(a.getBytes()) + " (" + GPUtils.byteArrayToReadableString(a.getBytes()) + ")");
323                                                        }
324                                                        System.out.println();
325                                                }
326                                        }
327                                }
328                        } catch (GPException e) {
329                                // All GP exceptions halt the program unless it is run with -relax
330                                if (!args.has(OPT_RELAX)) {
331                                        e.printStackTrace();
332                                        System.exit(1);
333                                }
334                                e.printStackTrace();
335                        } catch (CardException e) {
336                                // Card exceptions skip to the next reader, if available and allowed
337                                if (args.has(OPT_CONTINUE)) {
338                                        continue;
339                                } else {
340                                        e.printStackTrace();
341                                        throw e; // No catch.
342                                }
343                        } finally {
344                                if (card != null)
345                                        TerminalManager.disconnect(card, true);
346                        }
347
348                }
349                System.exit(0);
350        }
351}
