001package openkms.gpj;
002
003import java.util.Arrays;
004
005
006public final  class TLVUtils {
007
008        static short skipTag(byte [] data, short offset, byte tag) {
009                if (data[offset] == tag)
010                        ++offset;
011                return offset;
012        }
013
014        static short skip_tag_or_throw(byte [] data, short offset, byte tag) {
015                short skip = skipTag(data, offset, tag);
016                if (skip == offset)
017                        throw new RuntimeException("Expected tag " + Integer.toHexString(tag) + " but had " + Integer.toHexString(data[offset]));
018                return skip;
019        }
020
021        static short get_length(byte[] data, short offset) {
022                return (short) (data[offset] & 0x00FF);
023        }
024
025        static short skipLength(byte[] data, short offset) {
026                return ++offset;
027        }
028
029        static int get_byte_value(byte[] data, short offset) {
030                return (short)(data[offset] & 0x00FF);
031        }
032
033        static short expectTag(byte[] data, short offset, byte tag) {
034                if (data[offset] == tag)
035                        ++offset;
036                return offset;
037        }
038        static short skipTagAndLength(byte []data, short offset, byte tag) {
039                offset = expectTag(data, offset, tag);
040                offset = skipLength(data, offset);
041                return offset;
042        }
043
044        static short getTagLength(byte [] data, short offset) {
045                ++offset; // FIXME: jumpOverTag
046                return getLength(data, offset);
047        }
048
049        static int getTLVTag(byte [] data, short offset) {
050                return data[offset] & 0xFF;
051        }
052
053        static String getTLVValueAsHex(byte [] data, short offset) {
054                short len = getTagLength(data, offset);
055                return LoggingCardTerminal.encodeHexString(Arrays.copyOfRange(data, offset + 2, offset + 2 + len));
056        }
057
058        static byte[] getTLVValueAsBytes(byte [] data, short offset) {
059                short len = getTagLength(data, offset);
060                return Arrays.copyOfRange(data, offset + 2, offset + 2 + len);
061        }
062
063        static byte[] getTLVAsBytes(byte [] data, short offset) {
064                short len = getTagLength(data, offset);
065                return Arrays.copyOfRange(data, offset, offset + 2 + len);
066        }
067
068
069        static int getTLVValueOffset(byte [] data, short offset) {
070                // FIXME
071                return offset+2;
072        }
073        static short getTagLength(byte [] data, short offset, byte tag) {
074                offset = expectTag(data, offset, tag);
075                offset = skipLength(data, offset);
076                return offset;
077        }
078
079        static short getLength(byte [] data, short offset) {
080                return (short)(data[offset] & 0x00FF);
081        }
082
083        static short skipAnyTag(byte [] data, short offset) {
084                ++offset; // FIXME
085                return (short) (offset + getLength(data, offset) +1 );
086        }
087
088        static short findTag(byte [] data, short offset, byte tag) {
089                while (true) {
090                        if (data[offset] == tag) {
091                                return offset;
092                        } else {
093                                offset = skipAnyTag(data, offset);
094                        }
095                }
096        }
097
098        // Given a MSB byte array with a length, increment it by one.
099        public static void buffer_increment(byte[] buffer, short offset, short len) {
100                if (len < 1)
101                        return;
102                for (short i=(short) (offset+len-1); i >= offset; i--) {
103                        if (buffer[i] != (byte) 0xFF) {
104                                buffer[i]++;
105                                break;
106                        } else
107                                buffer[i] = (byte) 0x00;
108                }
109        }
