110        public void setKey(KeyType type, byte[] value) {
111                if (value.length < 16)
112                        throw new IllegalArgumentException("Key must be at least 16 bytes");
113                switch (type) {
114                case ENC:
115                        enc_key = value;
116                        break;
117                case MAC:
118                        mac_key = value;
119                        break;
120                case KEK:
121                        kek_key = value;
122                        break;
123                case RMAC:

125                        break;
126                default:
127                        break;

146        public byte[] get3DES(KeyType type) {
147                byte[] key24 = new byte[24];
148                System.arraycopy(getKey(type), 0, key24, 0, 16);
149                System.arraycopy(getKey(type), 0, key24, 16, 8);
150                return key24;
151        }

166        public boolean needsDiversity() {
167                return diversification != KeyDiversification.NONE && !diversified;
168        }
169        public void diversify(byte[] initialize_update_response) {
170                if (diversified || diversification == KeyDiversification.NONE) {
171                        throw new IllegalStateException("Already diversified or not needed!");
172                }
173
174                try {
175                        Cipher cipher = Cipher.getInstance("DESede/ECB/NoPadding");
176                        byte[] data = new byte[16];
177                        for (KeyType v : KeyType.values()) {
178                                if (v == KeyType.RMAC)
179                                        continue;
180
181                                // shift around and fill initialize update data as required.
182                                if (diversification == KeyDiversification.VISA2) {
183                                        fillVisa(data, initialize_update_response, v);
184                                } else if (diversification == KeyDiversification.EMV) {
185                                        fillEmv(data, initialize_update_response, v);
186                                }
187
188                                // Encrypt with current master key
189                                cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(get3DES(v), "DESede"));
190
191                                // Replace the key
192                                setKey(v, cipher.doFinal(data));
193                        }
194
195                        diversified = true;
196                } catch (BadPaddingException e) {
197                        throw new RuntimeException("Diversification failed.", e);
198                } catch (NoSuchAlgorithmException e) {
199                        throw new RuntimeException("Diversification failed.", e);
200                } catch (NoSuchPaddingException e) {
201                        throw new RuntimeException("Diversification failed.", e);
202                } catch (InvalidKeyException e) {
203                        throw new RuntimeException("Diversification failed.", e);
204                } catch (IllegalBlockSizeException e) {
205                        throw new RuntimeException("Diversification failed.", e);
206                }
207        }
208
209        private void fillVisa(byte[] data, byte[] init_update_response, KeyType key) {
210                System.arraycopy(init_update_response, 0, data, 0, 2);
211                System.arraycopy(init_update_response, 4, data, 2, 4);
212                data[6] = (byte) 0xF0;
213                data[7] = (byte) key.value;
214                System.arraycopy(init_update_response, 0, data, 8, 2);
215                System.arraycopy(init_update_response, 4, data, 10, 4);
216                data[14] = (byte) 0x0F;
217                data[15] = (byte) key.value;
218        }
219
220        private void fillEmv(byte[] data, byte[] init_update_response, KeyType key) {
221                // 6 rightmost bytes of init update response (which is 10 bytes)
222                System.arraycopy(init_update_response, 4, data, 0, 6);
223                data[6] = (byte) 0xF0;
224                data[7] = (byte) key.value;
225                System.arraycopy(init_update_response, 4, data, 8, 6);
226                data[14] = (byte) 0x0F;
227                data[15] = (byte) key.value;
228        }
