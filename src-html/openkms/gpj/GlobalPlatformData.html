019        public static String get_key_type_coding_string(int type) {
020                if ((0x00 <= type) && (type <= 0x7f))
021                        return "Reserved for private use";
022                if (0x80 == type)
023                        return "DES - mode (ECB/CBC) implicitly known";
024                if ((0x81 <= type) && (type <= 0x9F))
025                        return "RFU (symmetric algorithms)";
026                if (0xA0 == type)
027                        return "RSA Public Key - public exponent e component (clear text)";
028                if (0xA1 == type)
029                        return "RSA Public Key - modulus N component (clear text)";
030                if (0xA2 == type)
031                        return "RSA Private Key - modulus N component";
032                if (0xA3 == type)
033                        return "RSA Private Key - private exponent d component";
034                if (0xA4 == type)
035                        return "RSA Private Key - Chinese Remainder P component";
036                if (0xA5 == type)
037                        return "RSA Private Key - Chinese Remainder Q component";
038                if (0xA6 == type)
039                        return "RSA Private Key - Chinese Remainder PQ component";
040                if (0xA7 == type)
041                        return "RSA Private Key - Chinese Remainder DP1 component";
042                if (0xA8 == type)
043                        return "RSA Private Key - Chinese Remainder DQ1 component";
044                if ((0xA9 <= type) && (type <= 0xFE))
045                        return "RFU (asymmetric algorithms)";
046                if (0xFF == type)
047                        return "Not Available";
048
049                return "UNKNOWN";
050        }

053        // Print the key template
054        public static void pretty_print_key_template(List<KeySet.Key> list, PrintStream out) {
055                boolean factory_keys = false;
056                out.flush();
057                for (Key k: list) {
058                        out.println("Key ID:" + k.getID() + " VER:" + k.getVersion() + " LEN:" + k.getLength());
059                        if (k.getVersion() == 0x00|| k.getVersion() == 0xFF)
060                                factory_keys = true;
061                }
062                if (factory_keys)
063                        out.println("Key version suggests factory keys");
064                out.flush();
065        }

067        // GP 2.1.1 9.3.3.1
068        public static List<KeySet.Key> get_key_template_list(byte[] data, short offset) {
069
070                // Return empty list if no data from card.
071                // FIXME: not really a clean solution
072                if (data == null)
073                        return new ArrayList<Key>();
074
075                offset = TLVUtils.skip_tag_or_throw(data, offset, (byte) 0xe0);
076                offset = TLVUtils.skipLength(data, offset);
077
078                ArrayList<KeySet.Key> list = new ArrayList<Key>();
079                while (offset < data.length) {
080                        offset = TLVUtils.skipTag(data, offset, (byte) 0xC0);
081                        int component_len = offset + TLVUtils.get_length(data, offset);
082                        offset = TLVUtils.skipLength(data, offset);
083
084                        int id = TLVUtils.get_byte_value(data, offset);
085                        offset++;
086                        int version = TLVUtils.get_byte_value(data, offset);
087                        offset++;
088                        while (offset < component_len) {
089                                int type = TLVUtils.get_byte_value(data, offset);
090                                offset++;
091                                int length = TLVUtils.get_byte_value(data, offset);
092                                offset++;
093                                list.add(new Key(version, id, length, type));
094                                break; // FIXME:
095                        }
096                }
097                return list;
098        }

101        // GP 2.1.1: F.2 Table F-1
102        public static void pretty_print_card_data(byte[] data, PrintStream out) {
103                if (data == null) {
104                        out.println("NO CARD DATA");
105                        return;
106                }
107                try {
108                        short offset = 0;
109                        offset = TLVUtils.skipTagAndLength(data, offset, (byte) 0x66);
110                        offset = TLVUtils.skipTagAndLength(data, offset, (byte) 0x73);
111                        while (offset < data.length) {
112                                int tag = TLVUtils.getTLVTag(data, offset);
113                                if (tag == 0x06) {
114                                        String oid = ASN1ObjectIdentifier.fromByteArray(TLVUtils.getTLVAsBytes(data, offset)).toString();
115                                        if (oid.equals("1.2.840.114283.1"))
116                                                out.println("GlobalPlatform card");
117                                } else if (tag == 0x60) {
118                                        out.println("Version: " + gp_version_from_tlv(data, offset));
119                                } else if (tag == 0x63) {
120                                        out.println("TAG3: " + ASN1ObjectIdentifier.fromByteArray(TLVUtils.getTLVValueAsBytes(data, offset)));
121                                } else if (tag == 0x64) {
122                                        out.println("SCP version: " + gp_scp_version_from_tlv(data, offset));
123                                } else if (tag == 0x65) {
124                                        out.println("TAG5: " + ASN1ObjectIdentifier.fromByteArray(TLVUtils.getTLVValueAsBytes(data, offset)));
125                                } else if (tag == 0x66) {
126                                        out.println("TAG6: " + ASN1ObjectIdentifier.fromByteArray(TLVUtils.getTLVValueAsBytes(data, offset)));
127                                } else {
128                                        out.println("Unknown tag: " + Integer.toHexString(tag));
129                                }
130                                offset = TLVUtils.skipAnyTag(data, offset);
131                        }
132                } catch (IOException e) {
133                        throw new RuntimeException(e);
134                }
135        }

137        private static String gp_version_from_tlv(byte[] data, short offset) {
138                try {
139                        String oid;
140                        oid = ASN1ObjectIdentifier.fromByteArray(TLVUtils.getTLVValueAsBytes(data, offset)).toString();
141                        if (oid.startsWith("1.2.840.114283.2")) {
142                                return oid.substring("1.2.840.114283.2.".length());
143                        } else {
144                                return "unknown";
145                        }
146                } catch (IOException e) {
147                        return "error";
148                }
149        }

165        public static void get_global_platform_version(byte[] data) {
166                short offset = 0;
167                offset = TLVUtils.skipTagAndLength(data, offset, (byte) 0x66);
168                offset = TLVUtils.skipTagAndLength(data, offset, (byte) 0x73);
169                offset = TLVUtils.findTag(data, offset, (byte) 0x60);
170        }

173        public static KeyDiversification suggestDiversification(byte[] cplc) {
174                if (cplc != null) {
175                        // G&D
176                        if (cplc[7] == 0x16 && cplc[8] == 0x71)
177                                return KeyDiversification.EMV;
178                        // TODO: Gemalto
179                }
180                return KeyDiversification.NONE;
181        }

189        public static void pretty_print_cplc(byte [] data, PrintStream out) {
190                if (data == null) {
191                        out.println("NO CPLC");
192                        return;
193                }
194                if (data.length < 3 || data[2] != 0x2A)
195                        throw new IllegalArgumentException("CPLC must be 0x2A bytes long");
196                short offset = 3;
197                //offset = TLVUtils.skipTag(data, offset, (short)0x9F7F);
198                out.println("***** Card CPLC:");
199                out.println("IC Fabricator: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
200                out.println("IC Type: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
201                out.println("Operating System ID: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
202                out.println("Operating System release date: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
203                out.println("Operating System release level: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
204                out.println("IC Fabrication Date: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
205                out.println("IC Serial Number: " + bytesAsHex(data, offset, offset + 4)); offset += 4;
206                out.println("IC Batch Identifier: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
207                out.println("IC Module Fabricator: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
208                out.println("IC Module Packaging Date: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
209                out.println("ICC Manufacturer: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
210                out.println("IC Embedding Date: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
211                out.println("IC Pre-Personalizer: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
212                out.println("IC Pre-Perso. Equipment Date: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
213                out.println("IC Pre-Perso. Equipment ID: " + bytesAsHex(data, offset, offset + 4)); offset += 4;
214                out.println("IC Personalizer: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
215                out.println("IC Personalization Date: " + bytesAsHex(data, offset, offset + 2)); offset += 2;
216                out.println("IC Perso. Equipment ID: " + bytesAsHex(data, offset, offset + 4));  offset += 4;
217        }
