031        public static String byteArrayToReadableString(byte[] array) {
032                if (array == null) {
033                        return "(null)";
034                }
035                StringBuffer s = new StringBuffer();
036                for (int i = 0; i < array.length; i++) {
037                        char c = (char) array[i];
038                        s.append(((c >= 0x20) && (c < 0x7f)) ? (c) : ("."));
039                }
040                return "|" + s.toString() + "|";
041        }

047        public static byte[] stringToByteArray(String s) {
048                s = s.replaceAll(" ", "").replaceAll(":", "").replaceAll("\t", "");

056        private static byte[] pad80(byte[] text, int offset, int length) {
057                if (length == -1) {
058                        length = text.length - offset;
059                }
060                int totalLength = length;
061                for (totalLength++; (totalLength % 8) != 0; totalLength++) {
062                        ;
063                }
064                int padlength = totalLength - length;
065                byte[] result = new byte[totalLength];
066                System.arraycopy(text, offset, result, 0, length);
067                result[length] = (byte) 0x80;
068                for (int i = 1; i < padlength; i++) {
069                        result[length + i] = (byte) 0x00;
070                }
071                return result;
072        }
073
074        public static byte[] pad80(byte[] text) {
075                return pad80(text, 0, text.length);
076        }

078        public static byte[] mac_3des(byte[] key, byte[] text, byte[] cv)  {
079                return mac_3des(key, text, 0, text.length, cv);
080        }

082        private static byte[] mac_3des(byte[] key, byte[] text, int offset, int length, byte[] cv) {
083                if (length == -1) {
084                        length = text.length - offset;
085                }
086
087                try {
088                        Cipher cipher = Cipher.getInstance("DESede/CBC/NoPadding");
089                        cipher.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(getKey(key, 24), "DESede"), new IvParameterSpec(cv));
090                        byte[] result = new byte[8];
091                        byte[] res = cipher.doFinal(text, offset, length);
092                        System.arraycopy(res, res.length - 8, result, 0, 8);
093                        return result;
094                } catch (Exception e) {
095                        throw new RuntimeException("MAC computation failed.", e);
096                }
097        }

099        public static byte[] mac_des_3des(byte[] key, byte[] text, byte[] iv) {
100                return mac_des_3des(key, text, 0, text.length, iv);
101        }

103        private static byte[] mac_des_3des(byte[] key, byte[] text, int offset, int length, byte[] iv) {
104                if (length == -1) {
105                        length = text.length - offset;
106                }
107
108                try {
109
110                        Cipher cipher1 = Cipher.getInstance("DES/CBC/NoPadding");
111                        cipher1.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(getKey(key, 8), "DES"), new IvParameterSpec(iv));
112                        Cipher cipher2 = Cipher.getInstance("DESede/CBC/NoPadding");
113                        cipher2.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(getKey(key, 24), "DESede"), new IvParameterSpec(iv));
114
115                        byte[] result = new byte[8];
116                        byte[] temp;
117
118                        if (length > 8) {
119                                temp = cipher1.doFinal(text, offset, length - 8);
120                                System.arraycopy(temp, temp.length - 8, result, 0, 8);
121                                cipher2.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(getKey(key, 24), "DESede"), new IvParameterSpec(result));
122                        }
123                        temp = cipher2.doFinal(text, (offset + length) - 8, 8);
124                        System.arraycopy(temp, temp.length - 8, result, 0, 8);
125                        return result;
126                } catch (Exception e) {
127                        e.printStackTrace();
128                        throw new RuntimeException("MAC computation failed.", e);
129                }
130        }
131
132        public static byte[] getKey(byte[] key, int length) {
133                if (length == 24) {
134                        byte[] key24 = new byte[24];
135                        System.arraycopy(key, 0, key24, 0, 16);
136                        System.arraycopy(key, 0, key24, 16, 8);
137                        return key24;
138                } else {
139                        byte[] key8 = new byte[8];
140                        System.arraycopy(key, 0, key8, 0, 8);
141                        return key8;
142                }
143        }
